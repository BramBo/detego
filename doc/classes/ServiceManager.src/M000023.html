  def update(group, event, params)
    ServiceLogger.debug "Updating !:: #{group} #{event}".console_blue
    params.each do |k,v|
      ServiceLogger.debug  "[ #{k} |=> #{v} ]".console_purple
    end
    @read_only = "#{group}, #{event}, #{params}"
  end
