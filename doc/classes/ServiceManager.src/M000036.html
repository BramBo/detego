  def start()
    # this doesnt take an int as a port nr ?!
    ARGV << "-p"; ARGV << @port

    # Mongrel needs to be started in the project root dir
    Dir.chdir(LOAD_PATH) if CONTAINER_PATH == Dir.getwd

    self.status= "Running.."
    require 'config/boot'
    require 'commands/server'
  end
