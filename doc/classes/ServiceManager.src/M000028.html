  def get_rand_dimension_hash
    dimensions      = rand(5)+2
    dimension_size  = rand(5)+1    
    h = { :name     => "#{dimensions} Dimensions Hash"}
    
    dimension_size.times do |i|
      dimensions.times do |j|
        instance_eval(%{h#{"[:dimension]"*j} = {:name => "#{j}th dimension"}})
      end
    end
    h
  end
