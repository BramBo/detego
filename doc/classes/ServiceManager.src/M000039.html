  def start
    DRb.install_acl(ACL.new( %w[allow all] ))      
    @srv = DRb.start_service "druby://#{@running_on}:#{instance_variable_get("@port").to_i}", (ServiceProviderProxy.new())

    self.status="Running.."
    DRb.thread.join    
  end
