  def start
    begin 
      @serv = TCPServer.new(@port.to_i)    
    rescue => e
      puts e
      return false
    end    
    puts "#{self.class} started on port: #{@port}"
    while true
      @t = Thread.new(@serv.accept) do |session|
        prot  = session.addr[0]
        cmd   = session.gets.chomp

        begin
          query_print(session, cmd)
        rescue => e
          puts e
        end
        session.shutdown( how=1 )        
        session.close
      end
    end
  end
