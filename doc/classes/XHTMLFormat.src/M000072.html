  def header(hsh)
    @last_open +=  1 if (hsh[:list_open] || hsh[:key])    
    r           = "<h#{last_open}>#{(hsh[:key] || hsh[:list_open] || hsh[:list_close])}</h#{last_open}>"
    @last_open -=  1 if (hsh[:list_close] ||hsh[:key])
    r
  end
