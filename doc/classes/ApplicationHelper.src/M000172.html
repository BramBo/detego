  def status_to_icon(status)
    img = "<img class='service_status' src='"
    case status  
     when /started/i then
      img += "/images/started.png"
     when /run/i then
      img += "/images/running.png"
     when /stop/i then
      img += "/images/stopped.png"      
     when /boot/i then
      img += "/images/booting.png"        
     else
      img += "/images/stopped.png"          
    end
    img+"' alt='#{status}' title='#{status}' />"
  end
