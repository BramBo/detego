      def method_missing(method, *args, &block)
        required_set?

        begin 
          service = @container.find(@domain).find(@service)
          service.invoke(method, *args, &block) unless service.nil?
        rescue => ex
          ContainerLogger.error "#{ex} for service: #{@providee.name}", 2, 2, 2
        ensure
          @domain = @service = nil
        end      
      end
