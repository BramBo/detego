      def start_service()
        required_set?
        
        begin
          @container.find(@domain).find(@service).start()
        rescue Exception => ex
          ContainerLogger.error "Error starting service #{@domain}::#{@service}!".console_yellow, 1
          return "error;#{ex}"                    
        rescue => ex
          ContainerLogger.error "Error starting service #{@domain}::#{@service}!".console_yellow, 1
          return "error;Error starting service #{@domain}::#{@service}!"
        end
        "#{@service} started!"
      end
