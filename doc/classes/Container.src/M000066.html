  def add_domain(name)
    @domains[name] = @domains[name] || Domain.new(name, self)
    
    domains = @domains.sort{|a,b| "#{a[0]}"<=>"#{b[0]}"}
    @domains.clear
    domains.each do |k,v|
      @domains[k] = v
    end
    
    notify_observable_base(ObservableBase::DOMAIN_ADDED, {:domain => name})
    @domains[name]
  end
