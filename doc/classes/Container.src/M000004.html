  def remove(name=nil)
    if name.nil?
      @domains.each do |n, domain|
        domain.remove
        @domains.delete(n)      
      end
    else
      @domains[name].remove
      @domains.delete(name)
    end
    
    ContainerLogger.warn "Deleted domain #{name} (#{name.class})"
    true
  end
