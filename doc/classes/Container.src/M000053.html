  def shutdown!
    puts "\nShutting down..."
    @domains.each do |n, domain|
      domain.find(:all).each do |s, service|
        begin
          service.shutdown()  
         rescue Exception => e;  next;
         rescue => e;            next;
        end
        puts " v | #{s} shutdown."
      end
    end
    exit
  end
